How to use

MySQL database setup

CREATE DATABASE `security`;
GRANT ALL PRIVILEGES ON `security`.* TO 'security'@'localhost' IDENTIFIED BY 'security';
USE `security`;

DROP TABLE IF EXISTS `users`;
CREATE TABLE `users` (
    `id`       INT UNSIGNED,
    `username` VARCHAR(10),
    `password` VARCHAR(10),
    PRIMARY KEY (`id`)
);

INSERT INTO `users` VALUES
    (1, 'user',  'pass' ),
    (2, 'leigh', 'leigh')
;
